 #!/bin/bash

# Reload Bash configuration
echo "Sourcing Bash configuration..."
source ~/.bashrc

# List all Kubernetes Pods in all namespaces
echo "Getting all Kubernetes pods in all namespaces..."
kubectl get pods --all-namespaces

# Enable kubectl and minikube autocompletion for Bash
echo "Enabling kubectl autocompletion..."
source <(kubectl completion bash)

echo "Enabling minikube autocompletion..."
source <(minikube completion bash)

# Install bash-completion if needed
echo "Installing bash-completion..."
sudo apt install -y bash-completion

# Reload autocompletion for kubectl and minikube
echo "Reloading kubectl autocompletion..."
source <(kubectl completion bash)

echo "Reloading minikube autocompletion..."
source <(minikube completion bash)

echo "Setup complete!"
